# Logo System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Produce the `modern-python` org logo plus 3 exemplar project logos as a reproducible, generated SVG kit under `brand/`.

**Architecture:** Canonical geometry, color tokens, and an `fonttools`-based text outliner live once in `brand/build/` (Python). A generator composes every asset (icon, favicon, horizontal + stacked lockups, social card) from those primitives and writes SVG files into `brand/`. The committed SVGs are generated output; regenerating must be deterministic. Tests validate the builder functions' output structure (well-formed XML, viewBox, colors, accessibility attrs, expected elements).

**Tech Stack:** Python 3.11+, `uv` (deps), `pytest` (tests), `fonttools` (text→path), vendored JetBrains Mono (OFL). Assets are hand-free SVG generated by Python.

## Global Constraints

Copied verbatim from `docs/superpowers/specs/2026-06-27-logo-system-design.md`. Every task implicitly includes these.

- **viewBox** for all icon marks: `0 0 100 100`. Stroke width: **8**. `stroke-linecap`/`linejoin`: **round**.
- **Colors (light):** primary green `#356852`, gold `#c98a00`, teal `#2a9d8f`, amber `#c2722b`. **Dark:** green `#3f8064`, gold `#e0a300` (teal/amber unchanged).
- **Org mark geometry** (exact):
  - Outer snake (green): path `M20 35 L20 65 Q20 80 35 80 L65 80 Q80 80 80 65 L80 35 Q80 20 65 20 L35 20`; head circle `cx=35 cy=20 r=5.5`.
  - Middle snake (gold): path `M65 57 L65 43 Q65 35 57 35 L43 35 Q35 35 35 43 L35 57 Q35 65 43 65 L57 65`; head circle `cx=65 cy=57 r=5.5`.
  - Core (green): circle `cx=50 cy=50 r=5`.
- **Favicon:** two-band — outer green snake + middle gold snake, **core dropped**.
- **Project frame:** the single outer snake (+ head), colored by category. Monogram/glyph inside, optically centered (baseline tuned to optical center ≈ y=48.5 region).
- **Stack glyph** (templates), framework-colored: `rect x=37 y=39 w=26 h=6 rx=3` · `rect x=37 y=48 w=26 h=6 rx=3` · `rect x=37 y=57 w=26 h=6 rx=3`.
- **Category rule:** DI core = green frame + gold monogram; integrations = green frame + framework-colored letters; templates = green frame + framework-colored stack glyph; messaging = teal frame + gold monogram; utilities = amber frame + gold monogram.
- **Wordmark:** JetBrains Mono, weight ~600, **lowercase**, color = green (light) / `#3f8064` (dark). Primary lockup = horizontal command-line (`icon  modern-python`); secondary = stacked.
- **Light/dark in a single SVG:** embedded `<style>` with `@media (prefers-color-scheme: dark)` driving CSS custom properties (same technique as existing `docs/assets/modern-python-mark.svg`).
- **Accessibility:** every SVG has `role="img"` and an `aria-label`.
- **Brand casing in any copy:** `modern-di`, `that-depends`, `Litestar`, `FastStream`, `FastAPI`, `Typer`.
- **Storage:** `brand/` at repo root.

```python
from pathlib import Path
from functools import lru_cache
from fontTools.ttLib import TTFont
from fontTools.pens.svgPathPen import SVGPathPen

_FONT_PATH = Path(__file__).parent / "fonts" / "JetBrainsMono-SemiBold.ttf"

@lru_cache(maxsize=1)
def _font() -> TTFont:
    return TTFont(_FONT_PATH)

def _upm() -> int:
    return _font()["head"].unitsPerEm

def cap_height_scaled(size: float) -> float:
    f = _font()
    cap = getattr(f["OS/2"], "sCapHeight", 0) or int(0.7 * _upm())
    return cap * size / _upm()

def outline_text(text, size, *, x, baseline_y, anchor="start", color="#000000"):
    f = _font()
    glyphset = f.getGlyphSet()
    cmap = f.getBestCmap()
    hmtx = f["hmtx"]
    upm = _upm()
    s = size / upm

    parts, cursor = [], 0
    for ch in text:
        gname = cmap[ord(ch)]
        pen = SVGPathPen(glyphset)
        glyphset[gname].draw(pen)
        d = pen.getCommands()
        if d:
            parts.append(f'<path d="{d}" transform="translate({cursor},0)"/>')
        cursor += hmtx[gname][0]

    width = cursor * s
    ox = x - width / 2 if anchor == "middle" else x
    inner = "".join(parts)
    # font units are y-up; scale(s,-s) flips to SVG y-down, baseline at baseline_y
    g = (f'<g fill="{color}" transform="translate({ox:.3f},{baseline_y:.3f}) '
         f'scale({s:.5f},{-s:.5f})">{inner}</g>')
    return g, width
```

from brand.build import tokens as t

OUTER_SNAKE = "M20 35 L20 65 Q20 80 35 80 L65 80 Q80 80 80 65 L80 35 Q80 20 65 20 L35 20"
MIDDLE_SNAKE = "M65 57 L65 43 Q65 35 57 35 L43 35 Q35 35 35 43 L35 57 Q35 65 43 65 L57 65"
STROKE = 8

def svg(body, *, w=100, h=100, label, style=""):
    return (f'<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 {w} {h}" '
            f'role="img" aria-label="{label}">{style}{body}</svg>')

def theme_style(struct_light, accent_light, struct_dark, accent_dark):
    return ("<style>:root{"
            f"--struct:{struct_light};--accent:{accent_light}"
            "}@media (prefers-color-scheme: dark){:root{"
            f"--struct:{struct_dark};--accent:{accent_dark}"
            "}}</style>")

def _outer(stroke_color="var(--struct)"):
    return (f'<g stroke="{stroke_color}" stroke-width="{STROKE}">'
            f'<path d="{OUTER_SNAKE}"/>'
            f'<circle cx="35" cy="20" r="5.5" fill="{stroke_color}" stroke="none"/></g>')

def _middle(stroke_color="var(--accent)"):
    return (f'<g stroke="{stroke_color}" stroke-width="{STROKE}">'
            f'<path d="{MIDDLE_SNAKE}"/>'
            f'<circle cx="65" cy="57" r="5.5" fill="{stroke_color}" stroke="none"/></g>')

def org_icon():
    body = ('<g fill="none" stroke-linecap="round" stroke-linejoin="round">'
            f'{_outer()}{_middle()}'
            '<circle cx="50" cy="50" r="5" fill="var(--struct)" stroke="none"/>'
            '</g>')
    style = theme_style(t.GREEN, t.GOLD, t.GREEN_DARK, t.GOLD_DARK)
    return svg(body, label="Modern Python", style=style)

def org_favicon():
    body = ('<g fill="none" stroke-linecap="round" stroke-linejoin="round">'
            f'{_outer()}{_middle()}</g>')
    style = theme_style(t.GREEN, t.GOLD, t.GREEN_DARK, t.GOLD_DARK)
    return svg(body, label="Modern Python", style=style)

from brand.build.text import outline_text, cap_height_scaled

OPTICAL_CENTER = 48.5

def _frame(frame_color):
    return (f'<g fill="none" stroke-linecap="round" stroke-linejoin="round">'
            f'{_outer(frame_color)}</g>')

def project_monogram(initials, *, frame_color, ink, label):
    size = 30 if len(initials) <= 2 else 26
    baseline = OPTICAL_CENTER + cap_height_scaled(size) / 2
    glyphs, _ = outline_text(initials, size, x=50, baseline_y=baseline,
                             anchor="middle", color=ink)
    return svg(_frame(frame_color) + glyphs, label=label)

def project_template(*, frame_color, ink, label):
    bars = "".join(
        f'<rect x="37" y="{y}" width="26" height="6" rx="3"/>'
        for y in (39, 48, 57)
    )
    glyph = f'<g fill="{ink}">{bars}</g>'
    return svg(_frame(frame_color) + glyph, label=label)
```

ICON_SIZE_IN_LOCKUP = 80  # px of the embedded icon nested-svg

def icon_inner(kind, **kw):
    if kind == "org":
        return ('<g fill="none" stroke-linecap="round" stroke-linejoin="round">'
                f'{_outer()}{_middle()}'
                '<circle cx="50" cy="50" r="5" fill="var(--struct)" stroke="none"/></g>')
    if kind == "monogram":
        size = 30 if len(kw["initials"]) <= 2 else 26
        baseline = OPTICAL_CENTER + cap_height_scaled(size) / 2
        glyphs, _ = outline_text(kw["initials"], size, x=50, baseline_y=baseline,
                                 anchor="middle", color=kw["ink"])
        return _frame(kw["frame_color"]) + glyphs
    if kind == "template":
        bars = "".join(f'<rect x="37" y="{y}" width="26" height="6" rx="3"/>'
                       for y in (39, 48, 57))
        return _frame(kw["frame_color"]) + f'<g fill="{kw["ink"]}">{bars}</g>'
    raise ValueError(kind)

def _wordmark_style():
    return ("<style>:root{--ink:" + t.GREEN + "}"
            "@media (prefers-color-scheme: dark){:root{--ink:" + t.GREEN_DARK + "}}</style>")

def lockup_horizontal(icon_inner_svg, wordmark, label):
    fs = 46
    wm, w = outline_text(wordmark, fs, x=0, baseline_y=0, color="var(--ink)")
    gap = 26
    icon_px = 96
    total_w = icon_px + gap + w
    h = 100
    baseline = h / 2 + cap_height_scaled(fs) / 2
    body = (
        f'<svg x="0" y="2" width="{icon_px}" height="{icon_px}" '
        f'viewBox="0 0 100 100">{icon_inner_svg}</svg>'
        f'<g transform="translate({icon_px + gap},{baseline})">'
        f'{wm.replace("translate(0.000,0.000)", "translate(0,0)")}</g>'
    )
    style = theme_style(t.GREEN, t.GOLD, t.GREEN_DARK, t.GOLD_DARK).rstrip("</style>") \
        .replace("<style>", "<style>") + ""  # keep struct/accent for the icon
    # combine icon theme + wordmark ink into one style block
    full_style = ("<style>:root{"
                  f"--struct:{t.GREEN};--accent:{t.GOLD};--ink:{t.GREEN}"
                  "}@media (prefers-color-scheme: dark){:root{"
                  f"--struct:{t.GREEN_DARK};--accent:{t.GOLD_DARK};--ink:{t.GREEN_DARK}"
                  "}}</style>")
    return svg(body, w=round(total_w, 1), h=h, label=label, style=full_style)

def lockup_stacked(icon_inner_svg, wordmark, label):
    fs = 22
    wm, w = outline_text(wordmark, fs, x=64, baseline_y=128, anchor="middle",
                         color="var(--ink)")
    icon_px = 96
    body = (f'<svg x="16" y="0" width="{icon_px}" height="{icon_px}" '
            f'viewBox="0 0 100 100">{icon_inner_svg}</svg>{wm}')
    full_style = ("<style>:root{"
                  f"--struct:{t.GREEN};--accent:{t.GOLD};--ink:{t.GREEN}"
                  "}@media (prefers-color-scheme: dark){:root{"
                  f"--struct:{t.GREEN_DARK};--accent:{t.GOLD_DARK};--ink:{t.GREEN_DARK}"
                  "}}</style>")

def social_card(icon_inner_svg, wordmark, label):

from pathlib import Path
from brand.build import geometry as g

ROOT = Path(__file__).resolve().parents[2]
ORG = ROOT / "brand" / "org"

def _write(path: Path, content: str):
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(content + "\n", encoding="utf-8")

def render_org():
    inner = g.icon_inner("org")
    _write(ORG / "icon.svg", g.org_icon())
    _write(ORG / "favicon.svg", g.org_favicon())
    _write(ORG / "horizontal.svg", g.lockup_horizontal(inner, "modern-python", label="Modern Python"))
    _write(ORG / "stacked.svg", g.lockup_stacked(inner, "modern-python", label="Modern Python"))
    _write(ORG / "social.svg", g.social_card(inner, "modern-python", label="Modern Python"))

def main():
    render_org()
    # project rendering added in later tasks

if __name__ == "__main__":
    main()
```

```python
from brand.build import tokens as t

PROJECTS = ROOT / "brand" / "projects"

def render_project(slug, inner_kind, wordmark, label, **icon_kw):
    base = PROJECTS / slug
    inner = g.icon_inner(inner_kind, **icon_kw)
    if inner_kind == "monogram":
        standalone = g.project_monogram(icon_kw["initials"],
                                        frame_color=icon_kw["frame_color"],
                                        ink=icon_kw["ink"], label=label)
    else:
        standalone = g.project_template(frame_color=icon_kw["frame_color"],
                                        ink=icon_kw["ink"], label=label)
    _write(base / "icon.svg", standalone)
    _write(base / "horizontal.svg", g.lockup_horizontal(inner, wordmark, label=label))
    _write(base / "stacked.svg", g.lockup_stacked(inner, wordmark, label=label))
```
Then in `main()` add:

                   initials="di", frame_color=t.GREEN, ink=t.GOLD)
